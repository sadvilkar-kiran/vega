package main

import (
	"log"
	"${APP_NAME}/data"
	"${APP_NAME}/handlers"
	"${APP_NAME}/middleware"
	"os"

	"github.com/sadvilkar-kiran/vega"
)

func initApplication() *application {
	path, err := os.Getwd()
	if err != nil {
		log.Fatal(err)
	}

	// init vega
	vegaApp := &vega.Vega{}
	err = vegaApp.New(path)
	if err != nil {
		log.Fatal(err)
	}

	vegaApp.AppName = "${APP_NAME}"

	myMiddleware := &middleware.Middleware{
		App: vegaApp,
	}

	myHandlers := &handlers.Handlers{
		App: vegaApp,
	}

	app := &application{
		App:        vegaApp,
		Handlers:   myHandlers,
		Middleware: myMiddleware,
	}

	app.App.Routes = app.routes()

	app.Models = data.New(app.App.DB.Pool)
	myHandlers.Models = app.Models
	app.Middleware.Models = app.Models

	return app
}
